75 inline static bool color_equal(const SDL_Color& a, const SDL_Color& b) {
76 return a.r == b.r && a.g == b.g && a.b == b.b && a.a == b.a;
77 }
78
79 inline static bool color_not_equal(const SDL_Color& a, const SDL_Color& b) {
80 return !color_equal(a, b);
81 }
82
83 // Optional: epsilon compare if you ever switch to float colors
84 inline static bool color_near(const SDL_Color& a, const SDL_Color& b, Uint8 tolerance = 1) {
85 return (std::abs(int(a.r) - int(b.r)) <= tolerance &&
86 std::abs(int(a.g) - int(b.g)) <= tolerance &&
87 std::abs(int(a.b) - int(b.b)) <= tolerance &&
88 std::abs(int(a.a) - int(b.a)) <= tolerance);
89 }
90
91 inline static std::string color_to_string(const SDL_Color& c) {
92 return "(" + std::to_string(c.r) + ", " +
93 std::to_string(c.g) + ", " +
94 std::to_string(c.b) + ", " +
95 std::to_string(c.a) + ")";
96 }
