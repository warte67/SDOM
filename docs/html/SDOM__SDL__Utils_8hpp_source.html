74 inline static bool color_equal(const SDL_Color& a, const SDL_Color& b) {
75 return a.r == b.r && a.g == b.g && a.b == b.b && a.a == b.a;
76 }
77
78 inline static bool color_not_equal(const SDL_Color& a, const SDL_Color& b) {
79 return !color_equal(a, b);
80 }
81
82 // Optional: epsilon compare if you ever switch to float colors
83 inline static bool color_near(const SDL_Color& a, const SDL_Color& b, Uint8 tolerance = 1) {
84 return (std::abs(int(a.r) - int(b.r)) <= tolerance &&
85 std::abs(int(a.g) - int(b.g)) <= tolerance &&
86 std::abs(int(a.b) - int(b.b)) <= tolerance &&
87 std::abs(int(a.a) - int(b.a)) <= tolerance);
88 }
89
90 inline static std::string color_to_string(const SDL_Color& c) {
91 return "(" + std::to_string(c.r) + ", " +
92 std::to_string(c.g) + ", " +
93 std::to_string(c.b) + ", " +
94 std::to_string(c.a) + ")";
95 }
