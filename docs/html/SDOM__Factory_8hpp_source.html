8 * The SDOM_Factory class is a singleton responsible for creating and managing instances of
9 * various SDOM framework objects, including display objects, events, and other core components.
10 * It provides a centralized interface for object instantiation, ensuring consistent creation
11 * patterns and facilitating resource management. The Factory supports registration of custom
12 * object types, allowing for extensibility and flexibility in the types of objects that can
13 * be created. By encapsulating the object creation logic, the Factory promotes modularity and
14 * maintainability within the SDOM framework, making it easier to manage dependencies and
15 * lifecycle of objects throughout the application.

104 void addResource(const std::string& name, std::unique_ptr<T> obj)
105 {
106 if (resources.count(name)) {
107 // Optionally log: duplicate resource name
108 ERROR("Factory::addResource: Resource with name '" + name + "' already exists.");
109 return;
110 }
111 if (!obj || !obj->onInit()) {
112 // Optionally log: failed initialization
113 ERROR("Factory::addResource: Resource with name '" + name + "' failed to initialize.");
114 return;
115 }
116 resources[name] = std::move(obj);
117 }

144 {
