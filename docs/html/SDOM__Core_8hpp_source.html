274 // --- Tab Priority --- //
275 struct TabPriorityComparator {
276 bool operator()(const DisplayHandle& a, const DisplayHandle& b) const {
277 // If either handle is invalid, treat invalid handles as lower priority
278 // so they end up at the back of the ordering. Avoid throwing an error
279 // here because handles may become invalid between insertion and
280 // comparison (e.g., when objects are destroyed during traversal).
281 if (!a.isValid() && !b.isValid()) return false;
282 if (!a.isValid()) return true; // a < b
283 if (!b.isValid()) return false; // a >= b
284 return a->getTabPriority() < b->getTabPriority();
285 }
286 };
