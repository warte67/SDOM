86 // Controls how SDL synchronizes frame presentation with the display’s
87 // vertical refresh cycle. Values are:
88 // • -1 (SDL_RENDERER_VSYNC_ADAPTIVE)
89 // Adaptive vsync. Uses standard vsync when the frame rate can be
90 // maintained, but allows tearing if the engine falls behind to reduce
91 // visible stutter.
92 // • 0 (SDL_RENDERER_VSYNC_DISABLED)
93 // Disable vsync entirely. Presentation returns immediately, allowing
94 // the engine to run uncapped.
95 // • 1 (SDL_RENDERER_VSYNC)
96 // Standard vsync. Presentation blocks until the next vertical refresh.
97 // • 2, 3, …
98 // Attempt to synchronize with every 2nd, 3rd, etc. refresh interval.
99 // Support for these values varies across drivers and platforms.
100 // Note:
101 // Not all backends support every mode. After calling SDL_SetRenderVSync(),
102 // always check the return value to verify whether the requested mode was
103 // successfully applied.

416 // --- Tab Priority --- //
417 struct TabPriorityComparator {
418 bool operator()(const DisplayHandle& a, const DisplayHandle& b) const {
419 // If either handle is invalid, treat invalid handles as lower priority
420 // so they end up at the back of the ordering. Avoid throwing an error
421 // here because handles may become invalid between insertion and
422 // comparison (e.g., when objects are destroyed during traversal).
423 if (!a.isValid() && !b.isValid()) return false;
424 if (!a.isValid()) return true; // a < b
425 if (!b.isValid()) return false; // a >= b
426 return a->getTabPriority() < b->getTabPriority();
427 }
428 };
