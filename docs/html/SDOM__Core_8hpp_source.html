212 // --- Tab Priority --- //
213 struct TabPriorityComparator {
214 bool operator()(const DomHandle& a, const DomHandle& b) const {
215 // If either handle is invalid, treat invalid handles as lower priority
216 // so they end up at the back of the ordering. Avoid throwing an error
217 // here because handles may become invalid between insertion and
218 // comparison (e.g., when objects are destroyed during traversal).
219 if (!a.isValid() && !b.isValid()) return false;
220 if (!a.isValid()) return true; // a < b
221 if (!b.isValid()) return false; // a >= b
222 return a->getTabPriority() < b->getTabPriority();
223 }
224 };
