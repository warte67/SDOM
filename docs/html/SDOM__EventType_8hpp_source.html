7 *
8 * The SDOM_EventType class defines and manages the various types of events that can
9 * be dispatched and handled within the SDOM framework. It provides a flexible system
10 * for registering both predefined and custom event types, each with configurable
11 * propagation properties such as capturing, bubbling, target-only, and global scope.
12 * EventType instances are used to categorize and identify events throughout the event
13 * system, enabling robust event-driven programming and fine-grained control over event
14 * flow. By supporting both standard UI events (mouse, keyboard, window, etc.) and
15 * user-defined events, the EventType class forms the foundation for extensible and
16 * scalable event handling in SDOM-based applications.

163 explicit EventType(const std::string& name)
164 : name(name), captures_(true), bubbles_(true), targetOnly_(false), global_(false)
165 { registerEventType(name); }
166
167 explicit EventType(const std::string& name, bool captures, bool bubbles, bool targetOnly, bool global)
168 : name(name), captures_(captures), bubbles_(bubbles), targetOnly_(targetOnly), global_(global)
169 {
170 registerEventType(name);
171 }
172
173 const std::string& getName() const { return name; }
174
175 bool operator==(const EventType& other) const { return name == other.name; }
176 bool operator!=(const EventType& other) const { return name != other.name; }
177 bool operator<(const EventType& other) const { return name < other.name; }
178
179 static void registerEventType(const std::string& name) {
180 if (registry.find(name) == registry.end()) {
181 registry.insert(name);
182 }
183 }
184
185 static bool isRegistered(const std::string& name) {
186 return registry.find(name) != registry.end();
187 }
