101struct VariantStorage {
102 // Containers share ownership of child Variants via shared_ptr. Copies
103 // are cheap and keep the same node graph; moves transfer that graph
104 // intact. Mutations must go through shared_ptr to avoid raw-pointer
105 // aliasing; const views should remain read-only.

109 std::shared_ptr<void> ptr;
110 std::type_index type{typeid(void)};
111 std::string typeName;
112
113 bool operator==(const DynamicValue& o) const noexcept {
114 return ptr == o.ptr && type == o.type && typeName == o.typeName;
115 }

500// Hash support for Variant: best-effort, consistent with operator== for
501// the implemented cases. VariantHash is declared outside the class but
502// is a friend so it can access internal storage for pointer-based types.
503struct VariantHash {
504 size_t operator()(const Variant& v) const noexcept {
505 using std::size_t;
506 using std::hash;
507 switch (v.type()) {
508 case VariantType::Null: return 0x9e3779b97f4a7c15ULL;
509 case VariantType::Bool: return hash<bool>()(v.toBool());
510 case VariantType::Int: return hash<int64_t>()(v.toInt64());
511 case VariantType::Real: return hash<long double>()(v.toDouble());
512 case VariantType::String: return hash<std::string>()(v.toString());
513 case VariantType::Array: {
514 size_t h = 1469598103934665603ULL;
515 if (auto a = v.array()) {
516 for (const auto &e : *a) {
517 if (!e) continue;
518 size_t eh = operator()(*e);
519 h ^= eh + 0x9e3779b97f4a7c15ULL + (h<<6) + (h>>2);
520 }
521 }
522 return h;
523 }
524 case VariantType::Object: {
525 size_t h = 0x9e3779b9;
526 if (auto o = v.object()) {
527 for (const auto &kv : *o) {
528 if (!kv.second) continue;
529 size_t hk = hash<std::string>()(kv.first);
530 size_t hv = operator()(*kv.second);
531 // order-independent combine
532 h ^= hk + 0x9e3779b97f4a7c15ULL + (hv<<6) + (hv>>2);
533 h ^= hv + 0x9e3779b97f4a7c15ULL + (hk<<6) + (hk>>2);
534 }
535 h ^= o->size() + 0x9e3779b97f4a7c15ULL;
536 }
537 return h;
538 }
539 case VariantType::DisplayHandle: {
540 if (auto dh = v.displayHandle()) {
541 return hash<std::string>()(dh->getName());
542 }
543 return 0;
544 }
545 case VariantType::AssetHandle: {
546 if (auto ah = v.assetHandle()) {
547 size_t h1 = hash<std::string>()(ah->getName());
548 size_t h2 = hash<std::string>()(ah->getType());
549 size_t h3 = hash<std::string>()(ah->getFilename());
550 size_t h = h1;
551 h ^= h2 + 0x9e3779b97f4a7c15ULL + (h<<6) + (h>>2);
552 h ^= h3 + 0x9e3779b97f4a7c15ULL + (h<<6) + (h>>2);
553 return h;
554 }
555 return 0;
556 }
557 case VariantType::Dynamic: {
558 if (auto dv = std::get_if<VariantStorage::DynamicValue>(&v.storage_->data)) {
559 size_t p = std::hash<const void*>()(dv->ptr.get());
560 size_t t = dv->type.hash_code();
561 return p ^ (t + 0x9e3779b97f4a7c15ULL + (p<<6) + (p>>2));
562 }
563 return 0;
564 }
565 default: return 0;
566 }
