41// ─────────────────────────────────────────────────────────────────────────────
42// Internal storage node (shared so copies are cheap)
43// ─────────────────────────────────────────────────────────────────────────────
44struct VariantStorage {
45 // Use indirection for containers so Variant can be forward-declared
46 // without requiring a complete type at the point these aliases are
47 // instantiated. Using shared_ptr keeps ownership semantics simple and
48 // avoids changing external APIs dramatically at the call site; callers
49 // that iterate elements will now receive shared_ptr<Variant> elements
50 // and should dereference them.
51 using Array = std::vector<std::shared_ptr<Variant>>;
52 using Object = std::unordered_map<std::string, std::shared_ptr<Variant>>;
53 struct DynamicValue {
54 std::shared_ptr<void> ptr;
55 std::type_index type{typeid(void)};
56 std::string typeName;
57
58 bool operator==(const DynamicValue& o) const noexcept {
59 return ptr == o.ptr && type == o.type && typeName == o.typeName;
60 }
61 };

63 struct LuaRefValue {
64 sol::object obj;
65 lua_State* L = nullptr;
66
67 LuaRefValue() = default;
68 LuaRefValue(const sol::object& o)
69 : obj(o)
70 , L(o.lua_state())
71 {}
72 bool operator==(const LuaRefValue& o) const noexcept {
73 return L == o.L && obj == o.obj;
74 }
75 bool validFor(sol::state_view sv) const noexcept {
76 return L == sv.lua_state() && obj.valid();
77 }
78 };
79
80 // Note: Variant stores Lua references in a small wrapper (LuaRefValue)
81 // which keeps the `sol::object` and the originating `lua_State*` so
82 // accesses can validate the reference against the target state. This
83 // prevents returning a `sol::object` bound to a different `lua_State`.
84 //
85 // Tables are treated specially: by default they are converted (deep
86 // copied) into Variant `Array`/`Object` structures (this acts as a
87 // snapshot of the table contents). A global table storage mode allows
88 // callers to request that tables be stored as Lua references instead
89 // (i.e. kept as `LuaRefValue`) when desired.
90
91 using Data = std::variant<
92 std::monostate, // Null
93 bool,
94 int64_t,
95 double,
96 std::string,
97 Array,
98 Object,
99 DynamicValue,
100 LuaRefValue
101 >;
102
103 Data data{};
104};

316 // -------------------- Inline API usage notes --------------------
317 // The following short examples are intended to live close to the API so
318 // future readers can quickly understand common usage patterns.
319 //
320 // Registering a converter (type-based)
321 // ----------------------------------
322 // Variant::ConverterEntry ce;
323 // ce.toLua = [](const VariantStorage::DynamicValue& dv, sol::state_view L)->sol::object { ... };
324 // ce.fromVariant = [](const Variant& v)->std::shared_ptr<void> { ... };
325 // Variant::registerConverter<MyType>("MyTypeName", std::move(ce));
326 //
327 // Registering a converter by name only
328 // ------------------------------------
329 // Variant::registerConverterByName("RuntimeOnlyType", std::move(ce));
330 //
331 // Constructing dynamic Variants
332 // -----------------------------
333 // auto v = Variant::makeDynamic<MyType>(std::make_shared<MyType>(...));
334 // auto v2 = Variant::makeDynamic<MyType>(/* constructor args for MyType */);
335 // auto dv = Variant::makeDynamicValue<MyType>(shared_ptr<MyType>);
336 //
337 // Snapshotting & Lua table ownership
338 // ----------------------------------
339 // Variant::setTableStorageMode(Variant::TableStorageMode::KeepLuaRef);
340 // Variant luaRef = Variant::fromLuaObject(some_table);
341 // Variant copy = luaRef.snapshot(); // deep-copied Variant::Object/Array
342 // Variant::setTableStorageMode(Variant::TableStorageMode::Copy);
343 //
344 // VariantView read-only traversal
345 // -------------------------------
346 // Variant::VariantView view(variant_instance);
347 // if (view.isObject()) { if (const Variant* v = view.get("key")) { /* inspect v */ } }
348 // ------------------------------------------------------------------------------

467// Hash support for Variant: best-effort, consistent with operator== for
468// the implemented cases. VariantHash is declared outside the class but
469// is a friend so it can access internal storage for pointer-based types.
470struct VariantHash {
471 size_t operator()(const Variant& v) const noexcept {
472 using std::size_t;
473 using std::hash;
474 switch (v.type()) {
475 case VariantType::Null: return 0x9e3779b97f4a7c15ULL;
476 case VariantType::Bool: return hash<bool>()(v.toBool());
477 case VariantType::Int: return hash<int64_t>()(v.toInt64());
478 case VariantType::Real: return hash<long double>()(v.toDouble());
479 case VariantType::String: return hash<std::string>()(v.toString());
480 case VariantType::Array: {
481 size_t h = 1469598103934665603ULL;
482 if (auto a = v.array()) {
483 for (const auto &e : *a) {
484 if (!e) continue;
485 size_t eh = operator()(*e);
486 h ^= eh + 0x9e3779b97f4a7c15ULL + (h<<6) + (h>>2);
487 }
488 }
489 return h;
490 }
491 case VariantType::Object: {
492 size_t h = 0x9e3779b9;
493 if (auto o = v.object()) {
494 for (const auto &kv : *o) {
495 if (!kv.second) continue;
496 size_t hk = hash<std::string>()(kv.first);
497 size_t hv = operator()(*kv.second);
498 // order-independent combine
499 h ^= hk + 0x9e3779b97f4a7c15ULL + (hv<<6) + (hv>>2);
500 h ^= hv + 0x9e3779b97f4a7c15ULL + (hk<<6) + (hk>>2);
501 }
502 h ^= o->size() + 0x9e3779b97f4a7c15ULL;
503 }
504 return h;
505 }
506 case VariantType::Dynamic: {
507 if (auto dv = std::get_if<VariantStorage::DynamicValue>(&v.storage_->data)) {
508 size_t p = std::hash<const void*>()(dv->ptr.get());
509 size_t t = dv->type.hash_code();
510 return p ^ (t + 0x9e3779b97f4a7c15ULL + (p<<6) + (p>>2));
511 }
512 return 0;
513 }
514 case VariantType::LuaRef: {
515 if (auto lr = std::get_if<VariantStorage::LuaRefValue>(&v.storage_->data)) {
516 size_t Lp = std::hash<const void*>()(lr->L);
517 size_t tp = std::hash<int>()(static_cast<int>(lr->obj.get_type()));
518 return Lp ^ (tp + 0x9e3779b97f4a7c15ULL + (Lp<<6) + (Lp>>2));
519 }
520 return 0;
521 }
522 default: return 0;
523 }
524 }
