38/*
39================================================================================
40SDOM::IDataObject Contract
41
42Intent:
43 - IDataObject is the core interface for all data-driven objects in SDOM.
44 - Provides a flexible, dynamic property and command system for runtime manipulation, introspection, and automation.
45 - Enables serialization and deserialization of object state for scripting, configuration, and editor integration.
46
47Requirements:
48 1. Dynamic Properties
49 - Objects can register named properties with getter/setter functions.
50 - Properties are accessible for reading and writing at runtime.
51 - Properties can be introspected (listed, queried) for editor and scripting use.
52
53 2. Dynamic Commands
54 - Objects can register named commands (methods) callable at runtime.
55 - Commands can accept arguments and perform actions on the object.
56 - Commands are introspectable for automation and scripting.
57
58 3. Serialization/Deserialization
59 - Objects can serialize their state to a data format (now Lua tables).
60 - Objects can be initialized/deserialized from a data format (Lua tables).
61 - Serialization includes all registered properties and children (for hierarchical objects).
62
63 4. Editor/Scripting Integration
64 - Properties and commands are discoverable for external tools (editors, consoles, scripts).
65 - Supports runtime modification and automation.
66
67 5. Extensibility
68 - New properties and commands can be registered at runtime.
69 - Supports inheritance and composition for complex objects.
70
71Non-Requirements:
72 - Not responsible for rendering, event handling, or resource management (handled by other interfaces).
73 - Not tied to a specific serialization format (was JSON, now Lua; could be extended).
74`
75Summary:
76 IDataObject is a data-centric, introspectable, and extensible interface for SDOM objects, focused on:
77 - Dynamic property/command registration
78 - Runtime introspection and manipulation
79 - Serialization/deserialization for scripting, configuration, and editor integration
80================================================================================
81*/

334 static T lua_value_case_insensitive(const sol::table& tbl, const std::string& key, const T& default_value)
335 {
336 for (const auto& kv : tbl)
337 {
338 std::string k = kv.first.as<std::string>();
339 if (std::equal(k.begin(), k.end(), key.begin(), key.end(),
340 [](char a, char b) { return std::tolower(a) == std::tolower(b); }))
341 {
342 return kv.second.as<T>();
343 }
344 }
345 return default_value;

348 static sol::table ensure_sol_table(sol::state_view lua, const std::string& typeName)
349 {
350 sol::table global = lua.globals();
351 sol::object maybe = global.raw_get_or(typeName, sol::lua_nil);
352 if (maybe.valid() && maybe != sol::lua_nil)
353 {
354 // Many sol2 builds expose a usertype as a table/metatable; some
355 // report it as a userdata. In either case, fetch the table view
356 // from the global entry without replacing it.
357 try {
358 if (maybe.is<sol::table>()) {
359 return maybe.as<sol::table>();
360 }
361 } catch(...) {}
362 // Fallback: treat the existing global as a table-like and return it
363 try { return lua[typeName]; } catch(...) {}
364 }
365 // Not present: create and publish a fresh table
366 sol::table newTable = lua.create_table();
367 global[typeName] = newTable;
368 if (false)
369 {
370 std::cout << CLR::MAGENTA << "[LUA] Created table for type: "
371 << CLR::LT_MAGENTA << typeName << CLR::RESET << std::endl;
372 }
373 return newTable;
374 }

376 // Register a usertype and attach it to the given Lua table (idempotent)
377 template <typename T, typename BaseT>
378 static sol::usertype<T> register_usertype_with_table(sol::state_view lua, const std::string& typeName)
379 {
380 // Guard against re-registration across multiple call sites and
381 // avoid clobbering existing usertype tables. We only track a
382 // boolean flag per (lua_State*, typeName), not sol objects.
383 using StateMap = std::unordered_map<std::string, bool>;
384 static std::unordered_map<lua_State*, StateMap> s_registered;
385 lua_State* L = lua.lua_state();
386 auto& reg = s_registered[L];
387 auto it = reg.find(typeName);
388 if (it != reg.end() && it->second) {
389 // Already registered in this state; return a handle to the existing usertype table.
390 try { return lua[typeName]; } catch(...) {
391 // If the table is somehow missing, fall through to create below.
392 }
393 }
394
395 // If a table already exists and was marked as registered, reuse it.
396 sol::object existing = lua.globals().raw_get_or(typeName, sol::lua_nil);
397 if (existing.valid() && existing != sol::lua_nil) {
398 try {
399 sol::table t = lua[typeName];
400 sol::object flag = t.raw_get_or("__usertype_registered", sol::lua_nil);
401 if (flag.valid() && flag.is<bool>() && flag.as<bool>()) {
402 reg[typeName] = true;
403 return lua[typeName];
404 }
405 } catch(...) {}
406 }
407
408 // Create once (no constructor exposed to Lua by default).
409 auto userType = lua.new_usertype<T>(
410 typeName,
411 sol::no_constructor,
412 sol::base_classes, sol::bases<BaseT>()
413 );
414
415 // Mark on the usertype table itself so subsequent calls can detect it.
416 try {
417 sol::table utbl = lua[typeName];
418 utbl["__usertype_registered"] = true;
419 } catch(...) {}
420
421 if (false)
422 {
423 std::cout << CLR::YELLOW << "[LUA] Registered usertype: "
424 << CLR::LT_YELLOW << typeName << CLR::RESET << std::endl;
425 }
426
427 reg[typeName] = true;
428 return userType;
429 }

431 // Return both legacy per-type table (globals[typeName]) and canonical
432 // registry table (SDOM_Bindings[typeName]). Also ensures a proper
433 // usertype is registered with the provided BaseT.
434 template <typename T, typename BaseT>
435 static std::pair<sol::table, sol::table> ensure_type_tables(sol::state_view lua, const std::string& typeName)
436 {
437 // Ensure usertype with base relationship (idempotent)
438 (void)register_usertype_with_table<T, BaseT>(lua, typeName);
439
440 // Legacy/compatibility per-type table in globals
441 sol::table legacy = ensure_sol_table(lua, typeName);
442
443 // Canonical registry root: SDOM_Bindings
444 sol::table root;
445 try { root = lua["SDOM_Bindings"]; } catch(...) {}
446 if (!root.valid()) {
447 root = lua.create_table();
448 lua["SDOM_Bindings"] = root;
449 }
450
451 // Canonical per-type registry: SDOM_Bindings[typeName]
452 sol::table reg = root.raw_get_or(typeName, sol::lua_nil);
453 if (!reg.valid()) {
454 reg = lua.create_table();
455 root[typeName] = reg;
456 }
457
458 return { legacy, reg };
459 }

502 // ============================================================
503 // üß± Legacy Lua Binding Helpers
504 // ============================================================
505 virtual void _registerLuaBindings(const std::string& typeName, sol::state_view lua)
506 {
507 // 1. Make sure the per-type Lua table exists.
508 sol::table typeTable;
509 try {
510 typeTable = ensure_sol_table(lua, typeName);
511 } catch (...) {
512 return; // If Lua isn‚Äôt ready yet, fail silently.
513 }
514
515 // 2. Make sure we run this block only once per type.
516 sol::object registered = typeTable.raw_get_or("__idataobject_initialized", sol::lua_nil);
517 if (registered.valid() && registered.is<bool>() && registered.as<bool>()) {
518 return; // Already initialized; ancestors can still add keys later.
519 }
520 typeTable["__idataobject_initialized"] = true;
521
522 // 3. Add the base-level helpers that every IDataObject should expose.
523 auto set_if_absent = [](sol::table& tbl, const char* name, auto&& fn) {
524 sol::object cur = tbl.raw_get_or(name, sol::lua_nil);
525 if (!cur.valid() || cur == sol::lua_nil) {
526 tbl.set_function(name, std::forward<decltype(fn)>(fn));
527 }
528 };
529
530 set_if_absent(typeTable, "getName", [](IDataObject& self) { return self.getName(); });
531 set_if_absent(typeTable, "setName", [](IDataObject& self, const std::string& newName) { self.setName(newName); });
532
533 // if (DEBUG_REGISTER_LUA)
534 if (false)
535 {
536 std::string typeNameLocal = "IDataObject";
537 std::cout << CLR::CYAN << "Registered " << CLR::LT_CYAN << typeNameLocal
538 << CLR::CYAN << " Lua bindings for type: " << CLR::LT_CYAN
539 << typeName << CLR::RESET << std::endl;
540 }
541 }
