38/*
39================================================================================
40SDOM::IDataObject Contract
41
42Intent:
43 - IDataObject is the core interface for all data-driven objects in SDOM.
44 - Provides a flexible, dynamic property and command system for runtime manipulation, introspection, and automation.
45 - Enables serialization and deserialization of object state for scripting, configuration, and editor integration.
46
47Requirements:
48 1. Dynamic Properties
49 - Objects can register named properties with getter/setter functions.
50 - Properties are accessible for reading and writing at runtime.
51 - Properties can be introspected (listed, queried) for editor and scripting use.
52
53 2. Dynamic Commands
54 - Objects can register named commands (methods) callable at runtime.
55 - Commands can accept arguments and perform actions on the object.
56 - Commands are introspectable for automation and scripting.
57
58 3. Serialization/Deserialization
59 - Objects can serialize their state to a data format (now Lua tables).
60 - Objects can be initialized/deserialized from a data format (Lua tables).
61 - Serialization includes all registered properties and children (for hierarchical objects).
62
63 4. Editor/Scripting Integration
64 - Properties and commands are discoverable for external tools (editors, consoles, scripts).
65 - Supports runtime modification and automation.
66
67 5. Extensibility
68 - New properties and commands can be registered at runtime.
69 - Supports inheritance and composition for complex objects.
70
71Non-Requirements:
72 - Not responsible for rendering, event handling, or resource management (handled by other interfaces).
73 - Not tied to a specific serialization format (was JSON, now Lua; could be extended).
74`
75Summary:
76 IDataObject is a data-centric, introspectable, and extensible interface for SDOM objects, focused on:
77 - Dynamic property/command registration
78 - Runtime introspection and manipulation
79 - Serialization/deserialization for scripting, configuration, and editor integration
80================================================================================
81*/

117 template<typename T>
118 static T lua_value_case_insensitive(const sol::table& tbl, const std::string& key, const T& default_value)
119 {
120 for (const auto& kv : tbl)
121 {
122 std::string k = kv.first.as<std::string>();
123 if (std::equal(k.begin(), k.end(), key.begin(), key.end(),
124 [](char a, char b) { return std::tolower(a) == std::tolower(b); }))
125 {
126 return kv.second.as<T>();
127 }
128 }
129 return default_value;
130 }

135 static sol::table ensure_sol_table(sol::state_view lua, const std::string& typeName)
136 {
137 sol::table global = lua.globals();
138 sol::object maybe = global.raw_get_or(typeName, sol::lua_nil);
139 if (maybe.valid() && maybe != sol::lua_nil)
140 {
141 // Many sol2 builds expose a usertype as a table/metatable; some
142 // report it as a userdata. In either case, fetch the table view
143 // from the global entry without replacing it.
144 try {
145 if (maybe.is<sol::table>()) {
146 return maybe.as<sol::table>();
147 }
148 } catch(...) {}
149 // Fallback: treat the existing global as a table-like and return it
150 try { return lua[typeName]; } catch(...) {}
151 }
152 // Not present: create and publish a fresh table
153 sol::table newTable = lua.create_table();
154 global[typeName] = newTable;
155 if (false)
156 {
157 std::cout << CLR::MAGENTA << "[LUA] Created table for type: "
158 << CLR::LT_MAGENTA << typeName << CLR::RESET << std::endl;
159 }
160 return newTable;
161 }

163 // Register a usertype and attach it to the given Lua table (idempotent)
164 template <typename T, typename BaseT>
165 static sol::usertype<T> register_usertype_with_table(sol::state_view lua, const std::string& typeName)
166 {
167 // Guard against re-registration across multiple call sites and
168 // avoid clobbering existing usertype tables. We only track a
169 // boolean flag per (lua_State*, typeName), not sol objects.
170 using StateMap = std::unordered_map<std::string, bool>;
171 static std::unordered_map<lua_State*, StateMap> s_registered;
172 lua_State* L = lua.lua_state();
173 auto& reg = s_registered[L];
174 auto it = reg.find(typeName);
175 if (it != reg.end() && it->second) {
176 // Already registered in this state; return a handle to the existing usertype table.
177 try { return lua[typeName]; } catch(...) {
178 // If the table is somehow missing, fall through to create below.
179 }
180 }
181
182 // If a table already exists and was marked as registered, reuse it.
183 sol::object existing = lua.globals().raw_get_or(typeName, sol::lua_nil);
184 if (existing.valid() && existing != sol::lua_nil) {
185 try {
186 sol::table t = lua[typeName];
187 sol::object flag = t.raw_get_or("__usertype_registered", sol::lua_nil);
188 if (flag.valid() && flag.is<bool>() && flag.as<bool>()) {
189 reg[typeName] = true;
190 return lua[typeName];
191 }
192 } catch(...) {}
193 }
194
195 // Create once (no constructor exposed to Lua by default).
196 auto userType = lua.new_usertype<T>(
197 typeName,
198 sol::no_constructor,
199 sol::base_classes, sol::bases<BaseT>()
200 );
201
202 // Mark on the usertype table itself so subsequent calls can detect it.
203 try {
204 sol::table utbl = lua[typeName];
205 utbl["__usertype_registered"] = true;
206 } catch(...) {}
207
208 if (false)
209 {
210 std::cout << CLR::YELLOW << "[LUA] Registered usertype: "
211 << CLR::LT_YELLOW << typeName << CLR::RESET << std::endl;
212 }
213
214 reg[typeName] = true;
215 return userType;
216 }

218 // Return both legacy per-type table (globals[typeName]) and canonical
219 // registry table (SDOM_Bindings[typeName]). Also ensures a proper
220 // usertype is registered with the provided BaseT.
221 template <typename T, typename BaseT>
222 static std::pair<sol::table, sol::table> ensure_type_tables(sol::state_view lua, const std::string& typeName)
223 {
224 // Ensure usertype with base relationship (idempotent)
225 (void)register_usertype_with_table<T, BaseT>(lua, typeName);
226
227 // Legacy/compatibility per-type table in globals
228 sol::table legacy = ensure_sol_table(lua, typeName);
229
230 // Canonical registry root: SDOM_Bindings
231 sol::table root;
232 try { root = lua["SDOM_Bindings"]; } catch(...) {}
233 if (!root.valid()) {
234 root = lua.create_table();
235 lua["SDOM_Bindings"] = root;
236 }
237
238 // Canonical per-type registry: SDOM_Bindings[typeName]
239 sol::table reg = root.raw_get_or(typeName, sol::lua_nil);
240 if (!reg.valid()) {
241 reg = lua.create_table();
242 root[typeName] = reg;
243 }
244
245 return { legacy, reg };
246 }

248 // Convenience: ensure usertype + tables, and return the canonical
249 // registry table (SDOM_Bindings[typeName]).
250 template <typename T, typename BaseT>
251 static sol::table getTypeTable(sol::state_view lua, const std::string& typeName)
252 {
253 auto both = ensure_type_tables<T, BaseT>(lua, typeName);
254 return both.second; // canonical registry table

268 virtual void _registerLuaBindings(const std::string& typeName, sol::state_view lua)
269 {
270 // 1. Make sure the per-type Lua table exists.
271 sol::table typeTable;
272 try {
273 typeTable = ensure_sol_table(lua, typeName);
274 } catch (...) {
275 return; // If Lua isnt ready yet, fail silently.
276 }
277
278 // 2. Make sure we run this block only once per type.
279 sol::object registered = typeTable.raw_get_or("__idataobject_initialized", sol::lua_nil);
280 if (registered.valid() && registered.is<bool>() && registered.as<bool>()) {
281 return; // Already initialized; ancestors can still add keys later.
282 }
283 typeTable["__idataobject_initialized"] = true;
284
285 // 3. Add the base-level helpers that every IDataObject should expose.
286 auto set_if_absent = [](sol::table& tbl, const char* name, auto&& fn) {
287 sol::object cur = tbl.raw_get_or(name, sol::lua_nil);
288 if (!cur.valid() || cur == sol::lua_nil) {
289 tbl.set_function(name, std::forward<decltype(fn)>(fn));
290 }
291 };
292
293 set_if_absent(typeTable, "getName", [](IDataObject& self) { return self.getName(); });
294 set_if_absent(typeTable, "setName", [](IDataObject& self, const std::string& newName) { self.setName(newName); });
295
296 // if (DEBUG_REGISTER_LUA)
297 if (false)
298 {
299 std::string typeNameLocal = "IDataObject";
300 std::cout << CLR::CYAN << "Registered " << CLR::LT_CYAN << typeNameLocal
301 << CLR::CYAN << " Lua bindings for type: " << CLR::LT_CYAN
302 << typeName << CLR::RESET << std::endl;
303 }
