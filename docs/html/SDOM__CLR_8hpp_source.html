225 inline static void get_terminal_size(int& width, int& height) {
226 #if defined(_WIN32)
227 CONSOLE_SCREEN_BUFFER_INFO csbi;
228 GetConsoleScreenBufferInfo(GetStdHandle(STD_OUTPUT_HANDLE), &csbi);
229 width = (int)(csbi.srWindow.Right-csbi.srWindow.Left+1);
230 height = (int)(csbi.srWindow.Bottom-csbi.srWindow.Top+1);
231 #elif defined(__linux__) || defined(__APPLE__)
232 struct winsize w;
233 ioctl(fileno(stdout), TIOCGWINSZ, &w);
234 width = (int)(w.ws_col);
235 height = (int)(w.ws_row);
236 #endif // Windows/Linux
237 }
