91 // Convert 8-bit pattern {R:3,G:3,B:2} to RGB ANSI
92 inline static std::string fg_color(Byte clr) {
93 Uint8 r3 = (clr >> 5) & 0x7;
94 Uint8 g3 = (clr >> 2) & 0x7;
95 Uint8 b2 = clr & 0x3;
96
97 Uint8 r8 = static_cast<Uint8>((r3 * 255 + 3) / 7);
98 Uint8 g8 = static_cast<Uint8>((g3 * 255 + 3) / 7);
99 Uint8 b8 = static_cast<Uint8>((b2 * 255 + 1) / 3);
100
101 return fg_rgb(r8, g8, b8);
102 }

276 inline static void get_terminal_size(int& width, int& height) {
277#if defined(_WIN32)
278 CONSOLE_SCREEN_BUFFER_INFO csbi;
279 GetConsoleScreenBufferInfo(GetStdHandle(STD_OUTPUT_HANDLE), &csbi);
280 width = static_cast<int>(csbi.srWindow.Right - csbi.srWindow.Left + 1);
281 height = static_cast<int>(csbi.srWindow.Bottom - csbi.srWindow.Top + 1);
282#elif defined(__linux__) || defined(__APPLE__)
283 struct winsize w {};
284 ioctl(fileno(stdout), TIOCGWINSZ, &w);
285 width = static_cast<int>(w.ws_col);
286 height = static_cast<int>(w.ws_row);
287#endif
288 }
