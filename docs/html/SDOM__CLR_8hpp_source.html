242 inline static void get_terminal_size(int& width, int& height) {
243 #if defined(_WIN32)
244 CONSOLE_SCREEN_BUFFER_INFO csbi;
245 GetConsoleScreenBufferInfo(GetStdHandle(STD_OUTPUT_HANDLE), &csbi);
246 width = (int)(csbi.srWindow.Right-csbi.srWindow.Left+1);
247 height = (int)(csbi.srWindow.Bottom-csbi.srWindow.Top+1);
248 #elif defined(__linux__) || defined(__APPLE__)
249 struct winsize w;
250 ioctl(fileno(stdout), TIOCGWINSZ, &w);
251 width = (int)(w.ws_col);
252 height = (int)(w.ws_row);
253 #endif // Windows/Linux
254 }
