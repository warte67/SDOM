178 inline static const std::unordered_map<int, std::string> button_state_to_string = []{
179 std::unordered_map<int,std::string> m;

188 inline static const std::unordered_map<std::string, int> string_to_button_state = []{
189 std::unordered_map<std::string,int> m;
190 for (auto &p : button_state_pairs) {
191 std::string key = p.second;
192 std::transform(key.begin(), key.end(), key.begin(),
193 [](unsigned char c){ return static_cast<char>(std::tolower(c)); });
194 m.emplace(std::move(key), p.first);
195 }
196 return m;

222 inline static std::optional<ButtonState> button_state_from_name(const std::string &name)
223 {
224 std::string key = name;
225 std::transform(key.begin(), key.end(), key.begin(),
226 [](unsigned char c){ return static_cast<char>(std::tolower(c)); });
227 auto it = string_to_button_state.find(key);
228 if (it == string_to_button_state.end()) return std::nullopt;
229 return static_cast<ButtonState>(it->second);
