103struct Bounds
104{
105 float left = 0.0f;
106 float top = 0.0f;
107 float right = 0.0f;
108 float bottom = 0.0f;
109
110 float width() const { return std::abs(right - left); }
111 float height() const { return std::abs(bottom - top); }
112
113 bool operator==(const Bounds& other) const {
114 constexpr float epsilon = 0.0001f;
115 return std::abs(left - other.left) < epsilon &&
116 std::abs(top - other.top) < epsilon &&
117 std::abs(right - other.right) < epsilon &&
118 std::abs(bottom - other.bottom) < epsilon;
119 }
120
121 bool operator!=(const Bounds& other) const {
122 return !(*this == other);
123 }
124};
