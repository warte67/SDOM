158 // Register C++ std::function or lambda
159 template<typename Fn>
160 void registerFunction(const std::string& typeName,
161 const FunctionInfo& meta,
162 Fn&& fn)
163 {
164 std::scoped_lock lk(mutex_);
165 FunctionInfo m = meta;
166 m.callable = std::make_any<std::decay_t<Fn>>(std::forward<Fn>(fn));
167 m.callable_kind = FunctionInfo::CallableKind::CppCallable;
168 entries_[typeName].functions.push_back(std::move(m));
169 }
170
171 // Register C function pointer (for CAPI)
172 template<typename R, typename... Args>
173 void registerCFunctionPtr(const std::string& typeName,
174 const FunctionInfo& meta,
175 R(*fp)(Args...))
176 {
177 std::scoped_lock lk(mutex_);
178 FunctionInfo m = meta;
179 m.callable_kind = FunctionInfo::CallableKind::CFunctionPtr;
180 m.c_function_ptr = reinterpret_cast<std::uintptr_t>(fp);
181
182 if (m.c_name.empty()) {
183 m.c_name = "SDOM_CFN_" + typeName + "_" + m.name;
184 }
185
186 entries_[typeName].functions.push_back(std::move(m));
187 }
188
189 // Register meta-only function (no runtime callable)
190 void registerFunction(const std::string& typeName, const FunctionInfo& meta);

192 // Register getter/setter property
193 template<typename Getter, typename Setter = std::nullptr_t>
194 void registerProperty(const std::string& typeName,
195 const PropertyInfo& meta,
196 Getter&& getter,
197 Setter&& setter = nullptr)
198 {
199 std::scoped_lock lk(mutex_);
200
201 PropertyInfo p = meta;
202 p.getter = std::make_any<std::decay_t<Getter>>(std::forward<Getter>(getter));
203
204 if constexpr (!std::is_same_v<std::decay_t<Setter>, std::nullptr_t>) {
205 p.setter = std::make_any<std::decay_t<Setter>>(std::forward<Setter>(setter));
206 }
207
208 entries_[typeName].properties.push_back(std::move(p));
209 }
