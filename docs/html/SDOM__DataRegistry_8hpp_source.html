308 template<typename Return, typename Fn, typename ArgsTuple, std::size_t... I>
309 CAPI::CallResult invokeCallableImpl(Fn& fn,
310 const std::vector<CAPI::CallArg>& args,
311 std::index_sequence<I...>)
312 {
313 if constexpr (std::is_void_v<Return>) {
314 fn(ArgDecoder<std::tuple_element_t<I, ArgsTuple>>::decode(args[I])...);
315 return ResultEncoder<void>::encode();
316 } else {
317 Return value = fn(ArgDecoder<std::tuple_element_t<I, ArgsTuple>>::decode(args[I])...);
318 return ResultEncoder<Return>::encode(value);
319 }

322 template<typename Fn>
323 inline CAPI::GenericCallable makeGenericCallable(Fn&& fn)
324 {
325 using FnDecay = std::decay_t<Fn>;
326 using Traits = FunctionTraits<FnDecay>;
327 using ArgsTuple = typename Traits::ArgsTuple;
328 using Return = typename Traits::ReturnType;
329
330 constexpr std::size_t Arity = Traits::Arity;
331
332 auto callableStorage = std::make_shared<FnDecay>(std::forward<Fn>(fn));
333
334 return [callableStorage](const std::vector<CAPI::CallArg>& args) -> CAPI::CallResult {
335 if (args.size() != Arity) {
336 return CAPI::CallResult::Void();
337 }
338 auto& callable = *callableStorage;
339 return invokeCallableImpl<Return, FnDecay, ArgsTuple>(callable, args, std::make_index_sequence<Arity>{});
340 };

614template<typename FnStorage>
615inline void DataRegistry::registerGenericCallableLocked(const std::string& typeName,
616 FunctionInfo& info,
617 FnStorage& fn)
618{

620 return;
621 }
622
623 const std::string dispatchName = ensureDispatchName(typeName, info);
624 if (dispatchName.empty()) {
625 return;
626 }
627
628 auto generic = detail::makeGenericCallable(fn);
629 CAPI::registerCallable(dispatchName, std::move(generic));
630 info.callable_kind = FunctionInfo::CallableKind::GenericRuntime;
631 info.runtime_tag = "c_dispatch";
