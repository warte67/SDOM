301 /*
302 * Note: Anchor name normalization
303 * --------------------------------
304 * Canonicalization of user-provided anchor strings is performed by
305 * the helper function SDOM::normalizeAnchorString(const std::string&),
306 * implemented in `src/SDOM_Utils.cpp`.
307 *
308 * Behavior summary:
309 * - Strips whitespace and lower-cases the input.
310 * - Recognizes a single operator character joining two keywords
311 * (one of: '-', '|', '&', '+', ',') and returns the joined form
312 * using '_' as the separator (e.g. "top+left" -> "top_left").
313 * - Validates keyword pairs (e.g. disallows "top_bottom" or
314 * "left_right") and returns an empty string while emitting a
315 * WARNING on invalid inputs.
316 *
317 * The Factory and other parsers call this function before looking up
318 * entries in `stringToAnchorPoint_` so consumers can pass flexible
319 * anchor names (spaces, different joiners, mixed case) and still
320 * resolve to the canonical enum value.
321 */

437 struct Bounds
438 {
439 float left = 0.0f;
440 float top = 0.0f;
441 float right = 0.0f;
442 float bottom = 0.0f;
443
448 float width() const { return abs(right - left); }
449
454 float height() const { return abs(bottom - top); }
455
456 // Equality operator with epsilon for floating-point comparison
457 bool operator==(const Bounds& other) const {
458 constexpr float epsilon = 0.0001f;
459 return std::abs(left - other.left) < epsilon &&
460 std::abs(top - other.top) < epsilon &&
461 std::abs(right - other.right) < epsilon &&
462 std::abs(bottom - other.bottom) < epsilon;
463 }
464
465 bool operator!=(const Bounds& other) const {
466 return !(*this == other);
467 }
468 };
