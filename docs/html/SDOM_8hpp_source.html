259 /*
260 * Note: Anchor name normalization
261 * --------------------------------
262 * Canonicalization of user-provided anchor strings is performed by
263 * the helper function SDOM::normalizeAnchorString(const std::string&),
264 * implemented in `src/SDOM_Utils.cpp`.
265 *
266 * Behavior summary:
267 * - Strips whitespace and lower-cases the input.
268 * - Recognizes a single operator character joining two keywords
269 * (one of: '-', '|', '&', '+', ',') and returns the joined form
270 * using '_' as the separator (e.g. "top+left" -> "top_left").
271 * - Validates keyword pairs (e.g. disallows "top_bottom" or
272 * "left_right") and returns an empty string while emitting a
273 * WARNING on invalid inputs.
274 *
275 * The Factory and other parsers call this function before looking up
276 * entries in `stringToAnchorPoint_` so consumers can pass flexible
277 * anchor names (spaces, different joiners, mixed case) and still
278 * resolve to the canonical enum value.
279 */

395 struct Bounds
396 {
397 float left = 0.0f;
398 float top = 0.0f;
399 float right = 0.0f;
400 float bottom = 0.0f;
401
406 float width() const { return abs(right - left); }
407
412 float height() const { return abs(bottom - top); }
413
414 // Equality operator with epsilon for floating-point comparison
415 bool operator==(const Bounds& other) const {
416 constexpr float epsilon = 0.0001f;
417 return std::abs(left - other.left) < epsilon &&
418 std::abs(top - other.top) < epsilon &&
419 std::abs(right - other.right) < epsilon &&
420 std::abs(bottom - other.bottom) < epsilon;
421 }
422
423 bool operator!=(const Bounds& other) const {
424 return !(*this == other);
425 }
426 };
