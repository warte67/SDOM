164 /*
165 * Note: Anchor name normalization
166 * --------------------------------
167 * Canonicalization of user-provided anchor strings is performed by
168 * the helper function SDOM::normalizeAnchorString(const std::string&),
169 * implemented in `src/SDOM_Utils.cpp`.
170 *
171 * Behavior summary:
172 * - Strips whitespace and lower-cases the input.
173 * - Recognizes a single operator character joining two keywords
174 * (one of: '-', '|', '&', '+', ',') and returns the joined form
175 * using '_' as the separator (e.g. "top+left" -> "top_left").
176 * - Validates keyword pairs (e.g. disallows "top_bottom" or
177 * "left_right") and returns an empty string while emitting a
178 * WARNING on invalid inputs.
179 *
180 * The Factory and other parsers call this function before looking up
181 * entries in `stringToAnchorPoint_` so consumers can pass flexible
182 * anchor names (spaces, different joiners, mixed case) and still
183 * resolve to the canonical enum value.
184 */

300 struct Bounds
301 {
302 float left;
303 float top;
304 float right;
305 float bottom;
306
311 float width() const { return abs(right - left); }
312
317 float height() const { return abs(bottom - top); }
318
319 // Equality operator with epsilon for floating-point comparison
320 bool operator==(const Bounds& other) const {
321 constexpr float epsilon = 0.0001f;
322 return std::abs(left - other.left) < epsilon &&
323 std::abs(top - other.top) < epsilon &&
324 std::abs(right - other.right) < epsilon &&
325 std::abs(bottom - other.bottom) < epsilon;
326 }
327
328 bool operator!=(const Bounds& other) const {
329 return !(*this == other);
330 }
331 };
