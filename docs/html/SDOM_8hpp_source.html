181 /*
182 * Note: Anchor name normalization
183 * --------------------------------
184 * Canonicalization of user-provided anchor strings is performed by
185 * the helper function SDOM::normalizeAnchorString(const std::string&),
186 * implemented in `src/SDOM_Utils.cpp`.
187 *
188 * Behavior summary:
189 * - Strips whitespace and lower-cases the input.
190 * - Recognizes a single operator character joining two keywords
191 * (one of: '-', '|', '&', '+', ',') and returns the joined form
192 * using '_' as the separator (e.g. "top+left" -> "top_left").
193 * - Validates keyword pairs (e.g. disallows "top_bottom" or
194 * "left_right") and returns an empty string while emitting a
195 * WARNING on invalid inputs.
196 *
197 * The Factory and other parsers call this function before looking up
198 * entries in `stringToAnchorPoint_` so consumers can pass flexible
199 * anchor names (spaces, different joiners, mixed case) and still
200 * resolve to the canonical enum value.
201 */

317 struct Bounds
318 {
319 float left = 0.0f;
320 float top = 0.0f;
321 float right = 0.0f;
322 float bottom = 0.0f;
323
328 float width() const { return abs(right - left); }
329
334 float height() const { return abs(bottom - top); }
335
336 // Equality operator with epsilon for floating-point comparison
337 bool operator==(const Bounds& other) const {
338 constexpr float epsilon = 0.0001f;
339 return std::abs(left - other.left) < epsilon &&
340 std::abs(top - other.top) < epsilon &&
341 std::abs(right - other.right) < epsilon &&
342 std::abs(bottom - other.bottom) < epsilon;
343 }
344
345 bool operator!=(const Bounds& other) const {
346 return !(*this == other);
347 }
348 };
