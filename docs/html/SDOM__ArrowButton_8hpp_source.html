186 // 1. Load all IconButton (and IDisplayObject) fields first
187 IconButton::InitStruct::from_json(j, out);

189 // 2. Load ArrowButton-specific fields
190 if (auto it = j.find("direction"); it != j.end())
191 {
192 if (it->is_string())
193 {
194 std::string dir = it->get<std::string>();
195 std::transform(dir.begin(), dir.end(), dir.begin(), ::tolower);
196 auto mapIt = string_to_arrow_direction.find(dir);
197 if (mapIt != string_to_arrow_direction.end())
198 {
199 out.direction = mapIt->second;
200 }
201 }
202 else if (it->is_number_integer())
203 {
204 out.direction = static_cast<ArrowDirection>(it->get<int>());
205 }
206 }
207 }
